```python
from elite_fleet_orchestrator import (
    EliteOrchestrator, Task, AgentType, Priority
)

# Create orchestrator
orch = EliteOrchestrator(
    working_dir=Path("/your/project"),
    source_dir=Path("/source/repo")
)

# Define custom task
task = Task(
    id="my_task",
    name="My Integration",
    description="Detailed task description...",
    agent_type=AgentType.HYBRID,
    priority=Priority.HIGH,
    dependencies=["setup_task"],
    files_to_modify=["src/App.tsx"],
    max_retries=3,
    timeout=600
)

# Execute
orch.add_task(task)
await orch.execute()

# Get report
report = orch.generate_report()
orch.display_report()
```

async def validate_typescript_build(result: TaskResult) -> bool:
    """Ensure TypeScript compiles without errors"""

        "npm run build",

    id="frontend_task",
